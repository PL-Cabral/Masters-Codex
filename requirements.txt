<script>
    let currentUser = null;
    let currentTableId = null;
    let pollingInterval = null;
    let lastTableDataString = ""; // Evita o "Flickering" (piscar da tela)

    const classStats = {
        "Guerreiro": { hp: 20, str: 15, dex: 12, con: 14, int: 8, wis: 10, cha: 10 },
        "Mago": { hp: 12, str: 8, dex: 12, con: 10, int: 15, wis: 14, cha: 10 },
        "Assassino": { hp: 15, str: 10, dex: 15, con: 12, int: 10, wis: 10, cha: 14 }
    };

    const mobBase = {
        "Goblin": { hp: 10, str: 8, dex: 14, threat: 1 },
        "Orc": { hp: 20, str: 16, dex: 10, threat: 2 },
        "Slime": { hp: 15, str: 10, dex: 8, threat: 1 },
        "Aranha": { hp: 12, str: 12, dex: 16, threat: 1 }
    };

    let charCreationData = { points: 7, base: {} };

    // --- CONTROLE DA TELA DE LOGIN ---
    function toggleLoginFields() {
        const role = document.getElementById('role-select').value;
        const credBox = document.getElementById('table-credentials');
        const masterOpts = document.getElementById('master-options');
        const gmPassBox = document.getElementById('gm-password-box');
        
        if (role === 'player') {
            masterOpts.style.display = 'none';
            credBox.style.display = 'block';
            gmPassBox.style.display = 'none'; // Jogador não tem senha!
        } else {
            masterOpts.style.display = 'block';
            credBox.style.display = document.getElementById('has-existing-table').checked ? 'block' : 'none';
            gmPassBox.style.display = 'block';
        }
    }

    function toggleMasterTableFields() {
        document.getElementById('table-credentials').style.display = document.getElementById('has-existing-table').checked ? 'block' : 'none';
    }

    window.onload = toggleLoginFields;

    async function performLogin() {
        const role = document.getElementById('role-select').value;
        const username = document.getElementById('username').value;
        const tableId = document.getElementById('login-table-id').value;
        const password = document.getElementById('login-password').value;
        
        if(!username) return alert("Digite seu nome!");

        if(role === 'master') {
            if(document.getElementById('has-existing-table').checked && (!tableId || !password)) {
                return alert("Para carregar uma mesa, forneça o ID e a Senha do Mestre.");
            }
            const res = await fetch('/api/login/master', { method: 'POST', headers:{'Content-Type': 'application/json'}, body: JSON.stringify({username, table_id: tableId, password})});
            const data = await res.json();
            
            if(data.status === 'success') {
                currentUser = data;
                document.getElementById('login-screen').style.display = 'none';
                document.getElementById('vtt-screen').style.display = 'block';
                
                if (data.table_id) {
                    currentTableId = data.table_id;
                    document.getElementById('gm-tools').style.display = 'block';
                    startVTT();
                } else {
                    document.getElementById('gm-create-table-panel').style.display = 'block';
                    document.getElementById('action-panel').style.display = 'none';
                }
            } else {
                alert(data.message);
            }
        } else {
            if(!tableId) return alert("O ID da mesa é obrigatório para jogar.");
            charCreationData.temp = {username, tableId};

            const res = await fetch('/api/login/player', { method: 'POST', headers:{'Content-Type': 'application/json'}, body: JSON.stringify({username, table_id: tableId})});
            const data = await res.json();
            
            if(data.status === 'success') {
                currentUser = data;
                currentTableId = tableId;
                startVTT();
            } else if (data.status === 'needs_creation') {
                document.getElementById('login-screen').style.display = 'none';
                document.getElementById('char-creation-screen').style.display = 'block';
                loadBaseStats();
            } else {
                alert(data.message);
            }
        }
    }

    // --- CRIAÇÃO DE PERSONAGEM ---
    function loadBaseStats() {
        const c = document.getElementById('char-class').value;
        charCreationData.base = {...classStats[c]};
        charCreationData.points = 7;
        updateCharUI();
    }

    function changeStat(stat, val) {
        if(val > 0 && charCreationData.points > 0) {
            charCreationData.base[stat]++;
            charCreationData.points--;
        } else if (val < 0 && charCreationData.base[stat] > classStats[document.getElementById('char-class').value][stat]) {
            charCreationData.base[stat]--;
            charCreationData.points++;
        }
        updateCharUI();
    }

    function updateCharUI() {
        document.getElementById('points-left').innerText = charCreationData.points;
        document.getElementById('char-hp').innerText = charCreationData.base.hp;
        ['str','dex','con','int','wis','cha'].forEach(s => { document.getElementById(`stat-${s}`).innerText = charCreationData.base[s]; });
    }

    async function finalizeCharCreation() {
        const payload = {
            username: charCreationData.temp.username,
            table_id: charCreationData.temp.tableId,
            char_class: document.getElementById('char-class').value,
            hp: charCreationData.base.hp,
            attributes: charCreationData.base
        };

        const res = await fetch('/api/player/create_character', { method: 'POST', headers:{'Content-Type': 'application/json'}, body: JSON.stringify(payload)});
        const data = await res.json();
        
        if(data.status === 'success') {
            currentUser = data;
            currentTableId = data.table_id;
            
            // Corrige o Redirecionamento da UI
            document.getElementById('char-creation-screen').style.display = 'none';
            startVTT();
        }
    }

    // --- AÇÕES DO MESTRE (CRIAR MESA E MOBS) ---
    async function createTable() {
        const name = document.getElementById('new-table-name').value;
        const pass = document.getElementById('new-table-password').value;
        if(!name || !pass) return alert("Forneça Nome e Senha para a mesa!");
        
        const res = await fetch('/api/table/create', { method: 'POST', headers:{'Content-Type': 'application/json'}, body: JSON.stringify({table_name: name, master_id: currentUser.user_id, password: pass})});
        const data = await res.json();
        if(data.status === 'success') {
            alert(`Mesa Criada!\nID: ${data.table_id}\nSenha Mestre: ${pass}`);
            currentTableId = data.table_id;
            document.getElementById('gm-create-table-panel').style.display = 'none';
            document.getElementById('gm-tools').style.display = 'block';
            document.getElementById('action-panel').style.display = 'block';
            startVTT();
            loadMobBaseStats();
        }
    }

    function loadMobBaseStats() {
        const type = document.getElementById('mob-type').value;
        const level = parseInt(document.getElementById('mob-level').value) || 1;
        const base = mobBase[type];
        
        document.getElementById('mob-name').value = `${type} Nv.${level}`;
        document.getElementById('mob-hp').value = base.hp + (level * 5);
        document.getElementById('mob-str').value = base.str + Math.floor(level/2);
        document.getElementById('mob-dex').value = base.dex + Math.floor(level/2);
    }

    async function addMob() {
        const payload = {
            table_id: currentTableId,
            type: 'monster',
            monster_type: document.getElementById('mob-type').value,
            name: document.getElementById('mob-name').value,
            level: document.getElementById('mob-level').value,
            hp: document.getElementById('mob-hp').value,
            threat: mobBase[document.getElementById('mob-type').value].threat,
            attributes: {
                strength: document.getElementById('mob-str').value,
                dexterity: document.getElementById('mob-dex').value
            }
        };

        await fetch('/api/table/add_entity', { method: 'POST', headers:{'Content-Type': 'application/json'}, body: JSON.stringify(payload)});
        fetchTableDetails(); 
    }

    // --- VTT CORE E TEMPO REAL SMOOTH (NO FLICKER) ---
    function startVTT() {
        document.getElementById('login-screen').style.display = 'none';
        document.getElementById('vtt-screen').style.display = 'block';
        fetchTableDetails();
        if(pollingInterval) clearInterval(pollingInterval);
        pollingInterval = setInterval(fetchTableDetails, 3000);
    }

    async function fetchTableDetails() {
        if(!currentTableId) return;
        const res = await fetch(`/api/table/${currentTableId}`);
        const data = await res.json();

        if(data.status === 'success') {
            // O Segredo Anti-Flicker: Compara o hash da mesa. Só atualiza a tela se algo tiver mudado!
            const newDataString = JSON.stringify(data);
            if (newDataString === lastTableDataString) return;
            lastTableDataString = newDataString;

            document.getElementById('table-name-display').innerText = data.name;
            document.getElementById('table-id-display').innerText = data.id;

            const actorSelect = document.getElementById('action-actor');
            const targetSelect = document.getElementById('action-target');
            const currentActor = actorSelect.value;
            const currentTarget = targetSelect.value;

            actorSelect.innerHTML = '<option value="">Selecione...</option>';
            targetSelect.innerHTML = '<option value="">Nenhum / Área</option>';

            const pList = document.getElementById('players-list');
            const mList = document.getElementById('monsters-list');
            pList.innerHTML = ''; mList.innerHTML = '';

            const renderEntity = (ent, listEl) => {
                const li = document.createElement('li');
                li.style.marginBottom = '10px';
                li.style.background = '#2c2c2c';
                li.style.padding = '10px';
                li.style.borderRadius = '5px';
                li.style.borderLeft = ent.hp <= 0 ? '4px solid #444' : (ent.type === 'player' ? '4px solid #2ed573' : '4px solid #ff4757');
                
                const percent = Math.max(0, Math.min(100, (ent.hp / ent.max_hp) * 100));
                let status = ent.hp <= 0 ? '(Derrotado)' : '';
                
                let quickBtns = '';
                if (currentUser.role === 'master' && ent.hp > 0) {
                    quickBtns = `
                        <div>
                            <button class="quick-btn btn-dmg" onclick="quickAdjust('${ent.id}', 5, false)">-5 HP</button>
                            <button class="quick-btn btn-heal" onclick="quickAdjust('${ent.id}', 5, true)">+5 HP</button>
                        </div>
                    `;
                }

                li.innerHTML = `
                    <div style="display:flex; justify-content:space-between; align-items:center; margin-bottom:6px;">
                        <div>
                            <strong style="${ent.hp <= 0 ? 'text-decoration: line-through; color: #888;' : ''}">${ent.name}</strong> 
                            <span style="color:#aaa; font-size:12px; margin-left: 5px;">[${ent.class}] ${status}</span>
                        </div>
                        <div style="display:flex; align-items:center; gap: 10px;">
                            <span style="font-weight:bold; font-family:monospace;">${ent.hp}/${ent.max_hp}</span>
                            ${quickBtns}
                        </div>
                    </div>
                    <div style="background:#111; width:100%; height:10px; border-radius:5px; overflow:hidden;">
                        <div class="hp-bar" style="width:${percent}%; background:${percent < 30 ? '#ff4757' : (ent.type==='player' ? '#2ed573' : '#ff9f43')}"></div>
                    </div>
                `;
                listEl.appendChild(li);

                if(ent.hp > 0) {
                    if(currentUser.role === 'master' || (currentUser.role === 'player' && ent.id === currentUser.char_id)) {
                        actorSelect.innerHTML += `<option value="${ent.id}">${ent.name}</option>`;
                    }
                    targetSelect.innerHTML += `<option value="${ent.id}">${ent.name}</option>`;
                }
            };

            data.players.forEach(p => renderEntity(p, pList));
            data.monsters.forEach(m => renderEntity(m, mList));

            // Restaura seleção dos Dropdowns se ainda existirem
            if(currentActor && actorSelect.querySelector(`option[value="${currentActor}"]`)) actorSelect.value = currentActor;
            if(currentTarget && targetSelect.querySelector(`option[value="${currentTarget}"]`)) targetSelect.value = currentTarget;

            const logBox = document.getElementById('combat-log-box');
            logBox.innerHTML = data.logs.join('<br><br>');
            logBox.scrollTop = logBox.scrollHeight;
        }
    }

    async function performAction() {
        const payload = {
            table_id: currentTableId,
            attacker_id: document.getElementById('action-actor').value,
            target_id: document.getElementById('action-target').value,
            action_name: document.getElementById('action-type').value,
            dice_type: document.getElementById('action-dice').value
        };

        if(!payload.attacker_id) return alert("Selecione quem fará a ação!");

        await fetch('/api/combat/action', { method: 'POST', headers:{'Content-Type': 'application/json'}, body: JSON.stringify(payload)});
        lastTableDataString = ""; // Força atualização da tela
        fetchTableDetails(); 
    }

    async function quickAdjust(targetId, amount, isHeal) {
        if(!confirm(`Confirmar ${isHeal ? 'Cura' : 'Dano'} de ${amount}?`)) return;
        await fetch('/api/combat/quick_adjust', { 
            method: 'POST', 
            headers:{'Content-Type': 'application/json'}, 
            body: JSON.stringify({ table_id: currentTableId, target_id: targetId, amount: amount, is_heal: isHeal })
        });
        lastTableDataString = ""; 
        fetchTableDetails();
    }
</script>
